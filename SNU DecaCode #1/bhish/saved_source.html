    static long gcd(long big, long small) {
	long b = Math.max(big, small);
	long s = Math.min(big, small);
	if (s == 0)
	    return b;
	else
	    return gcd(s, b % s);
    }
    static long lcm(long a,long b){
	
	return (a * b) / gcd(a, b);
	
    }
    
    private static void solve(FastScanner s1, PrintWriter out){

	long n = s1.nextLong();
	long a = s1.nextLong();
	long b = s1.nextLong();
	long p = s1.nextLong();
	long q = s1.nextLong();
	long both = n / lcm(a, b);
	long red  = (n / a) - both;
	long blue = (n / b) - both;

	long max  = (red * p) + (blue * q) + (both * Math.max(p,q));
	out.print(max);
    }
